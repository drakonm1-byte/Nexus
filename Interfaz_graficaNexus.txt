ESPECIFICACIÓN COMPLETA DE LA INTERFAZ GRÁFICA DE NEXUS CON PYWEBVIEW
Para programadores: código listo para producción, integración con servicios en segundo plano y bandeja del sistema
1. VISIÓN GENERAL
La interfaz de Nexus es una aplicación web estática (HTML/CSS/JS) que se incrusta en una ventana nativa de Windows mediante PyWebView. Esto permite:

Desarrollo rápido: se programa la UI con tecnologías web estándar.

Comunicación bidireccional directa con el núcleo Python a través de un objeto js_api inyectado.

Bajo consumo de recursos (una sola ventana, sin procesos adicionales de Node.js).

Integración nativa con Windows (se puede usar la API del sistema, arrastrar archivos, etc.).

La ventana no se abre automáticamente al iniciar Nexus; solo aparece cuando el usuario la solicita desde el icono de la bandeja del sistema. Esto se gestiona mediante SystemTray (ver sección 5).

├── core/
│   ├── orchestrator.py         # Orquestador (con show_window())
│   ├── service_manager.py      # Gestiona Docker, Ollama en segundo plano

├── main.py                     # Punto de entrada que crea la ventana y el puente
└── assets/
    └── nexus_icon.ico          # Icono para la bandeja y la ventana
Nota: La carpeta webapp debe copiarse íntegramente en el instalador; PyInstaller la incluirá con --add-data "webapp;webapp".

// ------------------------------------------------------------
// Comunicación con el backend

function sendMessageToBackend(message) {
    if (window.nexusAPI && window.nexusAPI.send_message) {
        window.nexusAPI.send_message(message);
    } else {
        console.error('nexusAPI.send_message no disponible');
        addMessageToChat("Error: No se puede conectar con el asistente.", "assistant");
    }
}

// Función que será llamada desde Python para recibir resultados
window.receiveMessage = function(data) {
    if (data.error) {
        addMessageToChat(`Error: ${data.error}`, "assistant");
        return;
    }
    if (data.results && data.results.length) {
        data.results.forEach(result => {
            if (result.code) {
                addCodeMessage(result.code, "assistant");
                if (result.quality && result.quality.pass) {
                    addMessageToChat("✅ Código generado y verificado.", "assistant");
                } else {
                    addMessageToChat("⚠️ El código generado tiene problemas de calidad.", "assistant");
                }
            } else {
                addMessageToChat(result.message || "Operación completada.", "assistant");
            }
        });
    } else {
        addMessageToChat("Comando procesado.", "assistant");
    }
    // Actualizar otros componentes (cola de tareas, árbol de archivos, etc.) si se reciben datos adicionales
    if (data.task_queue) updateTaskQueue(data.task_queue);
    if (data.file_tree) updateFileTree(data.file_tree);
    if (data.provider_metrics) updateProviderMetrics(data.provider_metrics);
};

function addMessageToChat(text, sender) {
    const messageDiv = document.createElement('div');
    messageDiv.className = `message ${sender}`;
    const contentDiv = document.createElement('div');
    contentDiv.className = 'message-content';
    contentDiv.textContent = text;
    messageDiv.appendChild(contentDiv);
    chatMessages.appendChild(messageDiv);
    chatMessages.scrollTop = chatMessages.scrollHeight;
}

function addCodeMessage(code, sender) {
    const messageDiv = document.createElement('div');
    messageDiv.className = `message ${sender}`;
    const contentDiv = document.createElement('div');
    contentDiv.className = 'message-content';
    const codeBlock = document.createElement('div');
    codeBlock.className = 'code-block';
    const pre = document.createElement('pre');
    pre.textContent = code;
    codeBlock.appendChild(pre);
    contentDiv.appendChild(codeBlock);
    messageDiv.appendChild(contentDiv);
    chatMessages.appendChild(messageDiv);
    chatMessages.scrollTop = chatMessages.scrollHeight;
}

function sendUserMessage() {
    const text = userInput.value.trim();
    if (!text) return;
    addMessageToChat(text, "user");
    sendMessageToBackend(text);
    userInput.value = '';
    userInput.style.height = 'auto';
}

// Actualización de árbol de archivos (llamada desde Python)
// ------------------------------------------------------------
window.updateFileTree = function(treeData) {
    fileTreeDiv.innerHTML = '';
    if (!treeData || !treeData.children) {
        fileTreeDiv.innerHTML = '<div class="tree-placeholder">Abre un proyecto para ver los archivos</div>';
        return;
    }
    function renderTree(node, level) {
        const indent = '  '.repeat(level);
        const div = document.createElement('div');
        div.textContent = indent + node.name;
        if (node.children) {
            node.children.forEach(child => renderTree(child, level + 1));
        }
        fileTreeDiv.appendChild(div);
    }
    renderTree(treeData, 0);
};

window.updateTaskQueue = function(tasks) {
    if (!tasks || tasks.length === 0) {
        taskQueueList.innerHTML = '<div class="empty-queue">No hay tareas pendientes</div>';
        backgroundTaskStatusSpan.innerText = 'Fondo: 0 tareas';
        return;
    }
    taskQueueList.innerHTML = '';
    tasks.forEach(task => {
        const taskDiv = document.createElement('div');
        taskDiv.className = 'task-item';
        taskDiv.innerHTML = `
            <div class="task-title">${task.title}</div>
            <div class="task-progress"><div class="task-progress-fill" style="width: ${task.progress || 0}%;"></div></div>
            <div class="task-actions">
                <button class="pause-task" data-id="${task.id}">⏸️</button>
                <button class="cancel-task" data-id="${task.id}">❌</button>
            </div>
        `;
        taskQueueList.appendChild(taskDiv);
    });

"""
Punto de entrada de Nexus. Inicia el orquestador, los servicios en segundo plano,
la bandeja del sistema y crea la ventana solo cuando el usuario la solicita.
"""

import webview
import threading
import sys
from core.orchestrator import NexusOrchestrator

class NexusBridge:
    """Puente entre el frontend JS y el orquestador Python."""
    def __init__(self, orchestrator):
        self.orchestrator = orchestrator
        self.current_window = None

    def send_message(self, message: str):
        """Llamada desde JS para enviar un mensaje del usuario."""
        # Procesar en hilo separado para no bloquear la UI
        threading.Thread(target=self._process, args=(message,), daemon=True).start()

    def _process(self, message):
        try:
            result = self.orchestrator.process_request(message)
            # Enviar resultado al frontend
            if self.current_window:
                import json
                self.current_window.evaluate_js(f"window.receiveMessage({json.dumps(result)})")
        except Exception as e:
            if self.current_window:
                import json
                self.current_window.evaluate_js(f"window.receiveMessage({json.dumps({'error': str(e)})})")

    def start_listening(self):
        """Activar reconocimiento de voz continuo."""
        self.orchestrator.voice.start_continuous()

    def stop_listening(self):
        self.orchestrator.voice.stop_continuous()

    def pause_task(self, task_id=None):
        self.orchestrator.queue.pause(task_id)

    def cancel_task(self, task_id):
        self.orchestrator.queue.cancel(task_id)

    def clear_completed_tasks(self):
        self.orchestrator.queue.clear_completed()

    def set_quality_profile(self, profile):
        self.orchestrator.settings['development']['default_quality_profile'] = profile

    def set_personality(self, personality):
        self.orchestrator.personality.current = personality

    def undo(self):
        self.orchestrator.undo.undo_last()

    def redo(self):
        self.orchestrator.undo.redo_last()

    def save_all(self):
        self.orchestrator.save_state()

def create_window(bridge):
    """Crea y muestra la ventana de Nexus."""

        fullscreen=False

    bridge.current_window = window
    webview.start(debug=False, http_server=False)  # Ya tenemos el servidor Flask en el orquestador
    return window

import pystray
import threading
from PIL import Image

class SystemTray:
    def __init__(self, orchestrator):
        self.orchestrator = orchestrator
        self.icon = None

    def run(self):
        icon_path = "assets/nexus_icon.ico"
        image = Image.open(icon_path) if icon_path else Image.new('RGB', (64, 64), color='blue')
        menu = pystray.Menu(
            pystray.MenuItem("Abrir Nexus", self.on_open),
            pystray.MenuItem("Estado", self.on_status),
            pystray.MenuItem("Salir", self.on_quit)
        )
        self.icon = pystray.Icon("nexus", image, "Nexus Assistant", menu)
        self.icon.run()

    def on_open(self, icon, item):
        self.orchestrator.show_window()

    def on_status(self, icon, item):
        import win32api
        win32api.MessageBox(0, "Nexus funcionando.\nServicios: Docker, Ollama", "Estado", 0x40)

    def on_quit(self, icon, item):
        self.orchestrator.shutdown()
        if self.icon:
            self.icon.stop()
        import sys
        sys.exit(0)
Nota al pie 5:

El icono de la bandeja se ejecuta en un hilo separado.

Al hacer clic en "Abrir Nexus", se llama a orchestrator.show_window() que crea la ventana.

Al salir, se llama a shutdown(), que detiene los servicios externos y cierra la ventana.
